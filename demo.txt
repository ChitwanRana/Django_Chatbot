  <script>
    const chatBox = document.getElementById("chat-box");
    const userInput = document.getElementById("user-input");
    const sendBtn = document.getElementById("send-btn");
    const loader = document.getElementById("loader");
    let source = null;

    function addMessage(content, sender) {
      const msg = document.createElement("div");
      msg.classList.add("message", sender);

      const bubble = document.createElement("div");
      bubble.classList.add("bubble");
      bubble.textContent = content;

      msg.appendChild(bubble);
      chatBox.appendChild(msg);
      chatBox.scrollTop = chatBox.scrollHeight;
    }

    function sendMessage() {
      const message = userInput.value.trim();
      if (!message) return;

      addMessage(message, "user");
      userInput.value = "";
      loader.style.display = "block";

      if (source) source.close();

      source = new EventSource(`/stream/?message=${encodeURIComponent(message)}`);
      let botMessage = "";
      let botBubble = null;

      source.onmessage = (event) => {
        loader.style.display = "none";
        if (!botBubble) {
          const msg = document.createElement("div");
          msg.classList.add("message", "assistant");

          botBubble = document.createElement("div");
          botBubble.classList.add("bubble");
          msg.appendChild(botBubble);
          chatBox.appendChild(msg);
        }

        botMessage += event.data;
        botBubble.textContent = botMessage;
        chatBox.scrollTop = chatBox.scrollHeight;
      };

      source.onerror = () => {
        loader.style.display = "none";
        source.close();
      };
    }

    sendBtn.onclick = sendMessage;
    userInput.addEventListener("keydown", (e) => {
      if (e.key === "Enter") sendMessage();
    });
  </script>
